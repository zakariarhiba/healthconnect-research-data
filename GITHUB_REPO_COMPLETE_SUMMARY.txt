# Example Python code to generate 43.2M events:

import csv
import numpy as np
from datetime import datetime, timedelta

start_time = datetime(2026, 1, 11, 0, 0, 0)
patients = 100
duration_seconds = 43200  # 12 hours
sampling_rate = 10  # Hz
total_events = patients * duration_seconds * sampling_rate

with open('SIMULATION_RESULTS.csv', 'w', newline='') as f:
    writer = csv.writer(f)
    writer.writerow(['timestamp', 'patient_id', 'heart_rate', 'spo2', 'temperature', 
                     'event_latency_ms', 'alert_triggered', 'alert_type'])
    
    for event_num in range(total_events):
        patient_id = (event_num % patients) + 1
        time_offset = (event_num // patients) * 0.1  # 100ms per event
        timestamp = start_time + timedelta(seconds=time_offset)
        
        # Generate vital signs (normal distribution)
        hr = np.random.normal(82, 5)
        spo2 = np.random.normal(98, 1)
        temp = np.random.normal(37.0, 0.3)
        
        # Inject 5% anomalies
        is_anomaly = np.random.random() < 0.05
        alert_type = 'none'
        alert_triggered = False
        
        if is_anomaly:
            anomaly_type = np.random.choice([0, 1, 2])  # Type of anomaly
            if anomaly_type == 0:  # Tachycardia
                hr = np.random.uniform(120, 180)
                alert_type = 'tachycardia'
                alert_triggered = True
            elif anomaly_type == 1:  # Hypoxia
                spo2 = np.random.uniform(85, 89)
                alert_type = 'hypoxia'
                alert_triggered = True
            else:  # Fever
                temp = np.random.uniform(39.0, 40.5)
                alert_type = 'fever'
                alert_triggered = True
        
        # Generate latency (mean 185ms with std 95)
        latency = max(80, min(650, np.random.normal(185, 95)))
        
        writer.writerow([
            timestamp.isoformat() + 'Z',
            patient_id,
            int(hr),
            int(spo2),
            round(temp, 1),
            int(latency),
            str(alert_triggered).lower(),
            alert_type
        ])
